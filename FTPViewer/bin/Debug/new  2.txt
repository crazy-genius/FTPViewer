   private void loadPreConfFromXml()
        {
            foreach (XmlNode node in proxyXML.DocumentElement) // Проходимся по всем узлам XML
            {
                ProxyClass proxy = new ProxyClass();  // Переменная будет принимать значения узлов
                foreach (XmlNode child in node.ChildNodes) // Пройдемся по детям каждого узла
                {
                    switch (child.LocalName.ToLower())
                    {
                        case @"address":
                            proxy.ProxyServer= child.InnerText;
                            break;
                        case @"port":
                            proxy.ProxyPort = Convert.ToInt32(child.InnerText) ;
                            break;
                        case @"type":
                            proxy.ProxyMode = Convert.ToInt32(child.InnerText);
                            break;
                        case @"autent":
                            proxy.AutentRequired = child.InnerText;
                            break;
                        case @"login":
                            proxy.ProxyLogin = child.InnerText;
                            break;
                        case @"pass":
                            proxy.ProxyPassword = child.InnerText;
                            break;
                    }
                }
                proxySrvList.Add(proxy); // Занесем наш прокси сервер в список
                proxy = null; // Обнулим ссылку на прокси.
            }
